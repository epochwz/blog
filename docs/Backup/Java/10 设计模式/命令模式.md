**命令模式：**某个方法需要完成某个行为，但该行为的具体实现无法确定，必须等到执行该方法时才可以确定，因此需要将"处理行为"作为参数传入该方法
```java
package jwz;

import java.util.Arrays;

// 命令接口:封装不同的处理逻辑\[命令\]
interface Command{
// 封装“处理行为”
// target \[被处理的对象\]
void process\(int\[\] target\);

}

// Command实现类:实现打印功能
class PrintCommand implements Command{

public void process\(int\[\] target\){

System.out.println\("迭代输出目标数组的元素:" + Arrays.toString\(target\)\);

}

}

// Command实现类：实现加法功能
class AddCommand implements Command{

public void process\(int\[\] target\){

int sum = 0;

for \(int tmp : target \){

sum += tmp;

}

System.out.println\("数组元素的总和是:" + sum\);

}

}

// 测试：命令模式
public class CommandTest{

// 定义处理数组的方法：通过cmd参数\[不同实现类\]控制不同的处理逻辑

public void processArray\(int\[\] target , Command cmd\){

cmd.process\(target\);

}

public static void main\(String\[\] args\){

CommandTest pa = new CommandTest\(\);

// 被处理的数据

int\[\] target = {3, -4, 6, 4};

// 第一次处理数组，具体处理行为取决于PrintCommand

pa.processArray\(target , new PrintCommand\(\)\);

System.out.println\("------------------"\);

// 第二次处理数组，具体处理行为取决于AddCommand

pa.processArray\(target , new AddCommand\(\)\);

}

}
```
